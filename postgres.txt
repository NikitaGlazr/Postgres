CREATE OR REPLACE FUNCTION check_grade(score INT)
RETURNS VARCHAR AS $$
DECLARE
    grade VARCHAR;
BEGIN
    IF score >= 90 THEN
        grade := 'A';
    ELSIF score >= 80 THEN
        grade := 'B';
    ELSIF score >= 70 THEN
        grade := 'C';
    ELSE
        grade := 'D';
    END IF;
    
    RETURN grade;
END;
$$ LANGUAGE plpgsql;
